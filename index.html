function loadSpots() { const tableBody = document.getElementById('spotsTable'); const filtBand = document.getElementById('filterBand').value; const filtMode = document.getElementById('filterMode').value; fetch('/api/spots') .then(res => res.json()) .then(data => { if(data.error) { tableBody.innerHTML = `${data.error}`; return; } tableBody.innerHTML = ""; let count = 0; const now = new Date(); data.forEach(spot => { const freq = parseFloat(spot.frequency); // Filtri bande if(filtBand === "20m" && (freq < 14.0 || freq > 14.35)) return; if(filtBand === "40m" && (freq < 7.0 || freq > 7.3)) return; if(filtMode !== "all" && spot.mode !== filtMode) return; count++; if(count > 40) return; // Calcolo del tempo trascorso (Tempo relativo) let timeStr = "Attivo"; if(spot.spotTime) { const spotDate = new Date(spot.spotTime); // POTA restituisce stringhe ISO in UTC const diffMs = now - spotDate; const diffMins = Math.floor(diffMs / 60000); const diffSecs = Math.floor(diffMs / 1000); if (diffMins < 1) { timeStr = diffSecs <= 10 ? "Ora" : `${diffSecs}s fa`; } else { timeStr = `${diffMins}m fa`; } } // Conversione e formattazione frequenza in MHz (es. 14.074 MHz) const freqMhz = freq.toFixed(3); const row = ` ${timeStr} ${spot.activator} ${freqMhz} MHz ${spot.mode} ${spot.reference} `; tableBody.innerHTML += row; }); if(count === 0) { tableBody.innerHTML = `Nessun attivatore corrisponde ai filtri`; } }) .catch(err => { tableBody.innerHTML = `Errore caricamento dati`; }); }
